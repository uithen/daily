对任意数组项为数字的数组进行正序排列

			function arrSort(arr) {
				for (var i = 0; i < arr.length; i++) {
					for (var j = 0; j < arr.length - i - 1; j++) {	
						if (arr[j] > arr[j + 1]) {
							var temp;
							temp = arr[j];
							arr[j] = arr[j + 1];
							arr[j + 1] = temp;
						}
					}
				}
				return arr;
			}
			console.log(arrSort([11, 2, 12, 17, 13, 6])); //  [2, 6, 11, 12, 13, 17]
