        function getAvg() {
            var sum = 0,
                len = arguments.length;
            for(var i = 0; i < len; i++) {
                sum += arguments[i];
            }
            return sum /len;
        }
        console.log(getAvg(2,6));   // 4
        console.log(getAvg(11,12,13,11,12,13)); // 12
